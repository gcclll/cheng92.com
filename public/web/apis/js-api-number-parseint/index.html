function parseWithRadix(Z, radix = 10) {
  let mathInt = 0;

  const zLen = Z.length;

  const base = "a".codePointAt(0);
  for (let i = 0; i < zLen; i++) {
    let c = Z[i].toLowerCase();

    if (c >= "a" && c <= "z" && radix > 10) {
      c = 10 + c.codePointAt(0) - base;
    }

    mathInt += +c * Math.pow(radix, zLen - i - 1);
  }
  return mathInt;
}
